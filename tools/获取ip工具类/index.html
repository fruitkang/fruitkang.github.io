/**
 * 获取本机ip
 *
 * @return 本机ip列表
 */
public static List<String> getIp() {

    //1、获取本机下的所有网卡
    Collection<NetworkInterface> networkInterfaces = NetUtil.getNetworkInterfaces();
    if (CollectionUtil.isEmpty(networkInterfaces)) {
        throw new BaseRuntimeException("无法获取当前主机下的网卡");
    }
    networkInterfaces = networkInterfaces.stream().filter(networkInterface -> {
        try {
            return !networkInterface.isLoopback();
        } catch (SocketException e) {
            throw new BaseRuntimeException("获取本机ip发生SocketException", e);
        }
    }).collect(Collectors.toList());

    List<String> ips = Lists.newArrayList();

    networkInterfaces.forEach(networkInterface -> {
        //获取该网卡下的所有地址
        Enumeration<InetAddress> addresses = networkInterface.getInetAddresses();
        while (addresses.hasMoreElements()) {
            InetAddress inetAddress = addresses.nextElement();
            //过滤环回地址
            if (!inetAddress.isLoopbackAddress()) {
                if ("127.0.0.1".equalsIgnoreCase(inetAddress.getHostAddress())) {
                    continue;
                }
                if (inetAddress instanceof Inet6Address) {
                    continue;
                }
                if (inetAddress instanceof Inet4Address) {
                    ips.add(inetAddress.getHostAddress());
                }
            }
        }
    });
    return ips;
}




protected String getIpAddr() {
    String ipAddress = null;
    RequestAttributes requestAttributes = RequestContextHolder.getRequestAttributes();
    if (requestAttributes != null) {
        HttpServletRequest request = ((ServletRequestAttributes) requestAttributes).getRequest();
        String headerXForwardedFor = "x-forwarded-for";
        ipAddress = request.getHeader(headerXForwardedFor);
        String ipAddressUnknown = "unknown";
        if (ipAddress == null || ipAddress.length() == 0 || ipAddressUnknown.equalsIgnoreCase(ipAddress)) {
            String headerProxyClientIp = "Proxy-Client-IP";
            ipAddress = request.getHeader(headerProxyClientIp);
        }

        if (ipAddress == null || ipAddress.length() == 0 || ipAddressUnknown.equalsIgnoreCase(ipAddress)) {
            String headerWlProxyClientIp = "WL-Proxy-Client-IP";
            ipAddress = request.getHeader(headerWlProxyClientIp);
        }

        if (ipAddress == null || ipAddress.length() == 0 || ipAddressUnknown.equalsIgnoreCase(ipAddress)) {
            ipAddress = request.getRemoteAddr();
            String ipAddressOther = "0:0:0:0:0:0:0:1";
            String ipAddressLocalhost = "127.0.0.1";
            if (ipAddressLocalhost.equals(ipAddress) || ipAddressOther.equals(ipAddress)) {
                //根据网卡取本机配置的IP
                InetAddress inet = null;
                try {
                    inet = InetAddress.getLocalHost();
                } catch (UnknownHostException e) {
                    e.printStackTrace();
                }
                if (inet != null) {
                    ipAddress = inet.getHostAddress();
                }
            }
        }
    }

    //对于通过多个代理的情况，第一个IP为客户端真实IP,多个IP按照','分割
    //"***.***.***.***".length() = 15
    if (ipAddress != null && ipAddress.length() > 15) {
        if (ipAddress.indexOf(",") > 0) {
            ipAddress = ipAddress.substring(0, ipAddress.indexOf(","));
        }
    }

    return ipAddress;
}
